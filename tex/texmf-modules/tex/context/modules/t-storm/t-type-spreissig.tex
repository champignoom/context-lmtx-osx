  \definefontsynonym [SerifBoldItalic]	      [Preissig-Bold-Italic]
  \definefontsynonym [SerifOrnament]	      [Preissig-Ornament]

  \definefontsynonym [SerifExt]		   [Preissig-Roman-Ext]
  \definefontsynonym [SerifBoldExt]	   [Preissig-Bold-Ext]
  \definefontsynonym [SerifItalicExt]	   [Preissig-Italic-Ext]
  \definefontsynonym [SerifBoldItalicExt]  [Preissig-Bold-Italic-Ext]
